## Production prix marché fiable

### Algorithme
```javascript
function calculateMarketPrice(listing) {
  // 1. Filtrer comparables
  const comparables = findComparables({
    brand: listing.brand,
    model: listing.model,
    year: listing.year ± 2,
    mileage: listing.mileage ± 20%,
    country: listing.country,
    fuel_type: listing.fuel_type
  });
  
  // 2. Calculer statistiques
  const prices = comparables.map(c => c.price);
  const avgPrice = average(prices);
  const medianPrice = median(prices);
  const stdDev = standardDeviation(prices);
  
  // 3. Ajuster selon kilométrage
  const mileageAdjustment = calculateMileageAdjustment(
    listing.mileage,
    avgMileage(comparables)
  );
  
  // 4. Prix marché final
  const marketPrice = medianPrice + mileageAdjustment;
  
  return {
    market_price: marketPrice,
    avg_price: avgPrice,
    median_price: medianPrice,
    min_price: min(prices),
    max_price: max(prices),
    comparables_count: comparables.length,
    confidence_index: calculateConfidence(comparables.length, stdDev)
  };
}
```

### Critères comparables
- **Marque/Modèle** : Exact match
- **Année** : ± 2 ans
- **Kilométrage** : ± 20%
- **Pays** : Même pays
- **Carburant** : Même type
- **Statut** : Active uniquement

## Génération indice de confiance

### Facteurs
```javascript
function calculateConfidenceIndex(comparables) {
  const factors = {
    count: 0.4,        // 40% - Nombre de comparables
    recency: 0.2,      // 20% - Fraîcheur des données
    priceSpread: 0.2,  // 20% - Écart des prix (faible = confiance)
    location: 0.1,      // 10% - Proximité géographique
    source: 0.1        // 10% - Qualité sources
  };
  
  let score = 0;
  
  // Nombre de comparables
  if (comparables.length >= 20) score += 40;
  else if (comparables.length >= 10) score += 30;
  else if (comparables.length >= 5) score += 20;
  else score += 10;
  
  // Fraîcheur (données < 7 jours)
  const recentCount = comparables.filter(c => c.age < 7).length;
  score += (recentCount / comparables.length) * 20;
  
  // Écart des prix (coefficient de variation)
  const cv = stdDev(comparables.prices) / avg(comparables.prices);
  if (cv < 0.1) score += 20;
  else if (cv < 0.2) score += 15;
  else score += 10;
  
  // Proximité géographique
  // ... (calcul selon distance)
  
  // Qualité sources
  // ... (selon source_platform)
  
  return Math.min(100, Math.max(0, score));
}
```

## Détection anomalies

### Prix trop bas
```javascript
function detectLowPriceAnomaly(listing, marketPrice) {
  const threshold = 0.15; // 15% en dessous du marché
  const priceDiff = (marketPrice - listing.price) / marketPrice;
  
  if (priceDiff > threshold) {
    return {
      anomaly: 'low_price',
      severity: priceDiff > 0.3 ? 'high' : 'medium',
      reason: `Prix ${(priceDiff * 100).toFixed(1)}% en dessous du marché`,
      recommendation: 'Vérifier état véhicule, kilométrage, historique'
    };
  }
  
  return null;
}
```

```javascript
function detectHighPriceAnomaly(listing, marketPrice) {
  const threshold = 0.20; // 20% au-dessus du marché
  const priceDiff = (listing.price - marketPrice) / marketPrice;
  
  if (priceDiff > threshold) {
    return {
      anomaly: 'high_price',
      severity: priceDiff > 0.4 ? 'high' : 'medium',
      reason: `Prix ${(priceDiff * 100).toFixed(1)}% au-dessus du marché`,
      recommendation: 'Vérifier options, état exceptionnel, rareté'
    };
  }
  
  return null;
}
```
